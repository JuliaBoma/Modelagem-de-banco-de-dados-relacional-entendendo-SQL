/*cria um schema com o nome do schema*/ CREATE SCHEMA Clube_do_Livro; CREATE TABLE LIVROS ( ID_LIVRO INT NOT NULL, NOME_LIVRO VARCHAR (100) NOT NULL, AUTORIA VARCHAR (100) NOT NULL, ADITORIA VARCHAR (100) NOT NULL, CATEGORIA VARCHAR (100) NOT NULL, PRECO DECIMAL (5, 2) NOT NULL, PRIMARY KEY (ID_LIVRO) ); CREATE TABLE ESTOQUE ( ID_LIVRO INT NOT NULL, QTD_ESTOQUE INT NOT NULL, PRIMARY KEY (ID_LIVRO) ); CREATE TABLE VENDAS ( ID_PEDIDO INT NOT NULL, ID_VENDENDOR INT NOT NULL, ID_LIVRO INT NOT NULL, QTD_VENDIDA INT NOT NULL, DATA_VENDA DATE NOT NULL, PRIMARY KEY (ID_VENDENDOR, ID_PEDIDO) ); CREATE TABLE VENDEDORES ( ID_VENDENDOR INT NOT NULL, NOME_VENDEDOR VARCHAR (255) NOT NULL PRIMARY KEY (ID_VENDENDOR) );
